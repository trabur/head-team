ht.relayvisor()

RELAYVISOR uses Math.floor() and Math.random(). together they act as a sort of lock & key; which can be used to keep things unknown or unseen by others.

                 ----------
                 |        |
                 |  bob   |
                 |        |
                 ----------
                     ^
                     |
    <------------------------------->
     ^                             ^
     |                             |
 ----------                   ----------
 |        |                   |        |
 |  chad  |                   |  fred  |
 |        |                   |        |
 ----------                   ----------

let bob = ht.secret(4, "CODE")
let chad = "CSTREE"
let fred = "WAAPAWSA"
// --with--
let result = ht.relayvisor(bob, { from: chad, to: fred })
// --or--
let result = ht.relayvisor(bob, { from: fred, to: chad })

// CSTREE --> WAAPAWSA <-- CSTREE ... CODE
// --or--
// WAAPAWSA --> CSTREE <-- WAAPAWSA ... CODE
