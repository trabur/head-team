let plateId = "ABC" // license plate
let highway = "1234567890" // road
let id = "test"
let store = { hello: "world" }
function callback(value) {
  console.log('received:', value)
}
// --with--
ht.mobile(plateId) // vehicle
ht.lane(plateId, highway) // transport
ht.key(plateId, id)
ht.value(plateId, id)
ht.highway.post(plateId, callback)
// --or--
ht.move(plateId, highway).key(id).value(store).post(callback)
// --then--
ht.move(plateId, highway).key(id).find(callback)
